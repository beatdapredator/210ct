def is_prime(number,t<-3):
    if number =  0:
        result <- False
    else if number = 2:
        result <- True
    else if number = 1 or number % 2 = 0:
        return <-False
    else if t*t > number:
        result <- True
    else if number % t = 0:
       result <- False

    else:
        return is_prime(number,t+2)
