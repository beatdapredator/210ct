def high_low(arr,high,low):
      left <- 0
      right <- integer(length(arr))
      while left < <- right:
              middle <- integer((left + right) /2)
              if arr[middle] < <- high and arr[middle] > <- low:
                    return True
              else if arr[middle] > high:
                    right <- middle – 1
              else if arr[middle] < low:
                    left <- middle +1
     return False
